#!/usr/bin/env python3

import psutil

import time

import curses

from datetime import datetime

from collections import deque

import math

# Configurações

HISTORY_LENGTH = 60

GRAPH_CHARS = [' ', '▁', '▂', '▃', '▄', '▅', '▆', '▇', '█']

GRAPH_HEIGHT = 10

MIN_WIDTH = 80

REFRESH_RATE = 2

# Históricos

cpu_history = deque(maxlen=HISTORY_LENGTH)

ram_history = deque(maxlen=HISTORY_LENGTH)

def safe_addstr(win, y, x, text, attr=0):

    """Adiciona texto com verificação de limites"""

    max_y, max_x = win.getmaxyx()

    if y < 0 or y >= max_y or x >= max_x:

        return

    try:

        win.addstr(y, x, text[:max_x-x], attr)

    except curses.error:

        pass

def get_system_info():

    """Coleta métricas do sistema"""

    # CPU

    cpu_percent = psutil.cpu_percent(interval=1)

    cpu_history.append(cpu_percent)

   

    # Memória

    mem = psutil.virtual_memory()

    ram_percent = mem.percent

    ram_history.append(ram_percent)

   

    # Processos

    processes = []

    for proc in psutil.process_iter(['pid', 'name', 'cpu_percent', 'memory_percent']):

        try:

            processes.append(proc.info)

        except (psutil.NoSuchProcess, psutil.AccessDenied, psutil.ZombieProcess):

            pass

   

    return {

        'cpu': {

            'percent': cpu_percent,

            'count': psutil.cpu_count(),

            'history': list(cpu_history)

        },

        'memory': {

            'total_gb': mem.total / (1024**3),

            'used_gb': mem.used / (1024**3),

            'percent': ram_percent,

            'history': list(ram_history)

        },

        'processes': sorted(processes, key=lambda p: p['cpu_percent'], reverse=True)[:20],

        'time': datetime.now().strftime("%H:%M:%S")

    }

def draw_graph(stdscr, y, x, data, label, color_pair, width):

    """Desenha gráfico com escala fixa e legendas"""

    h, w = stdscr.getmaxyx()

    if y + GRAPH_HEIGHT + 2 >= h or x + width + 10 >= w:

        return

    current = data[-1] if data else 0

   

    # Escala fixa de 0-100%

    min_val = 0

    max_val = 100

    range_val = max_val - min_val

   

    # Moldura com legendas

    safe_addstr(stdscr, y, x, '┌' + '─' * width + '┐', curses.color_pair(color_pair))

   

    # Legendas verticais

    for i in range(GRAPH_HEIGHT + 1):

        value = 100 - (i * (100/GRAPH_HEIGHT))

        label_y = y + i + 1

        safe_addstr(stdscr, label_y, x-5, f"{value:3.0f}% ", curses.color_pair(color_pair))

        safe_addstr(stdscr, label_y, x, '│', curses.color_pair(color_pair))

        safe_addstr(stdscr, label_y, x + width + 1, '│', curses.color_pair(color_pair))

   

    safe_addstr(stdscr, y + GRAPH_HEIGHT + 1, x, '└' + '─' * width + '┘', curses.color_pair(color_pair))

    # Dados

    for i in range(min(width, len(data))):

        value = data[-(width - i)]

        bar_height = (value - min_val) * (GRAPH_HEIGHT / range_val)

        full_blocks = math.floor(bar_height)

        partial = math.floor((bar_height - full_blocks) * (len(GRAPH_CHARS)-1))

       

        for h_level in range(GRAPH_HEIGHT):

            char = GRAPH_CHARS[-1] if h_level < full_blocks else (

                  GRAPH_CHARS[partial] if h_level == full_blocks else ' ')

            safe_addstr(stdscr, y + GRAPH_HEIGHT - h_level, x + 1 + i, char,

                       curses.color_pair(color_pair))

    # Legenda inferior

    legend = f"{label}: {current:.1f}%"

    safe_addstr(stdscr, y + GRAPH_HEIGHT + 2, x, legend, curses.color_pair(color_pair))

def draw_info(stdscr, info):

    """Renderiza a interface"""

    stdscr.clear()

    h, w = stdscr.getmaxyx()

   

    if w < MIN_WIDTH or h < 24:

        msg = f"Terminal pequeno (min {MIN_WIDTH}x24). Atual: {w}x{h}"

        safe_addstr(stdscr, 0, (w - len(msg))//2, msg, curses.A_BOLD | curses.color_pair(1))

        return

   

    # Configura cores

    curses.init_pair(1, curses.COLOR_CYAN, curses.COLOR_BLACK)  # CPU

    curses.init_pair(2, curses.COLOR_GREEN, curses.COLOR_BLACK)  # RAM

   

    # Cabeçalho

    title = f" 🖥️  Linux Mint System Monitor [{info['time']}] "

    safe_addstr(stdscr, 0, (w - len(title))//2, title, curses.A_BOLD | curses.color_pair(1))

   

    # Gráficos

    graph_width = min(60, w - 15)

    start_y = 2

   

    # Gráfico de CPU

    draw_graph(stdscr, start_y, 8, info['cpu']['history'], "CPU", 1, graph_width)

   

    # Gráfico de Memória

    draw_graph(stdscr, start_y + GRAPH_HEIGHT + 5, 8, info['memory']['history'], "RAM", 2, graph_width)

   

    # Tabela de processos

    proc_x = graph_width + 15

    if proc_x < w - 25 and h > start_y + 10:

        safe_addstr(stdscr, start_y, proc_x, '╔════════╦════════════════╦══════╦══════╗')

        safe_addstr(stdscr, start_y+1, proc_x, '║  PID   ║ Nome           ║ CPU% ║ MEM% ║')

        safe_addstr(stdscr, start_y+2, proc_x, '╠════════╬════════════════╬══════╬══════╣')

       

        for i, proc in enumerate(info['processes'][:h-start_y-4]):

            line = f"║ {proc['pid']:6} ║ {proc['name'][:14]:14} ║ {proc['cpu_percent']:4.1f} ║ {proc['memory_percent']:4.1f} ║"

            safe_addstr(stdscr, start_y+3+i, proc_x, line)

       

        safe_addstr(stdscr, start_y+4+i, proc_x, '╚════════╩════════════════╩══════╩══════╝')

    # Rodapé

    footer = " Pressione 'Q' para sair ".center(w, '─')

    safe_addstr(stdscr, h-1, 0, footer, curses.A_DIM | curses.color_pair(1))

def main(stdscr):

    curses.curs_set(0)

    stdscr.nodelay(1)

   

    # Inicializa históricos

    for _ in range(HISTORY_LENGTH):

        cpu_history.append(0)

        ram_history.append(0)

   

    while True:

        info = get_system_info()

        draw_info(stdscr, info)

       

        key = stdscr.getch()

        if key == ord('q'):

            break

       

        time.sleep(REFRESH_RATE)

if __name__ == "__main__":

    try:

        curses.wrapper(main)

    except KeyboardInterrupt:

        print("\nMonitor encerrado pelo usuário.")
